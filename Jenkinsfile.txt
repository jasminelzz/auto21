pipeline {
    agent any
    stages {
        stage('Checkout') {
            steps {
                git 'https://github.com/autotestMTG21/auto21.git'
            }
        }
        stage('Build') {
            steps {
                bat "mvn compile"
            }
        }
        stage('Test') {
            steps {
                bat "mvn test"
            }
            post {
                always {
                    junit '**/TEST*.xml'
                }
            }
        }
         stage('newman') {
            steps {
                sh 'newman run Labb_Postman.postman_collection.json --environment Labb_Postman Enviroment.postman_environment.json --reporters junit'
            }
            post {
                always {
                        junit '**/*xml'
                    }
                }
        }
    }
}
